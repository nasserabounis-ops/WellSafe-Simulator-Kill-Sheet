// --- 1. STRICT API ROUNDING UTILITIES ---
const fixFloat = (val) => Math.round(val * 1e8) / 1e8;
const roundUp = (value, decimals = 1) => { 
  const multiplier = Math.pow(10, decimals);
  return Math.ceil(fixFloat(value * multiplier)) / multiplier;
};
const roundDown = (value, decimals = 0) => { 
  const multiplier = Math.pow(10, decimals);
  return Math.floor(fixFloat(value * multiplier)) / multiplier;
};
const roundNearest = (value) => Math.round(fixFloat(value)); 
const round1 = (value) => Math.round(fixFloat(value) * 10) / 10; 
const safeNum = (val) => Number(val) || 0;

  const elem = document.documentElement;

};

const InteractiveCell = ({ name, value, onChange, correctValue, isCalculated, w = 'w-16', suffix = '', decimals = null, mode, readOnlyStyle = false }) => {
  const displayValue = decimals !== null && correctValue !== undefined ? Number(correctValue).toFixed(decimals) : correctValue;
  
  if (mode === 'instructor' || isCalculated) {
    let bgClass = "bg-white border-black text-black"; 
    if (mode === 'student' && isCalculated) {
      const studentVal = Number(value);
      const expectedVal = Number(correctValue);
      const isEmpty = value === '' || value === undefined || value === null;
      
      let tolerance = 0.2;
      if (Math.abs(expectedVal) > 50) tolerance = 1;
      if (Math.abs(expectedVal) > 500) tolerance = 3;
      if (Math.abs(expectedVal) > 2000) tolerance = 5;

      const isCorrect = !isEmpty && Math.abs(studentVal - expectedVal) <= tolerance; 
      bgClass = isCorrect ? 'bg-emerald-100 text-emerald-900 border-emerald-500' : isEmpty ? 'bg-amber-100 text-amber-900 border-amber-500' : 'bg-rose-100 text-rose-900 border-rose-500';
    }
    return (
      <div className={`inline-flex items-center justify-center ${w} px-1 py-0.5 font-bold border-2 ${bgClass}`}>
        {mode === 'student' && isCalculated && value !== '' && Math.abs(Number(value) - Number(correctValue)) > 0.5 && (
          <span className="line-through text-[10px] mr-1 opacity-60 leading-none text-rose-700">{value}</span>
        )}
        <span>{displayValue}</span>
        {suffix && <span className="text-xs ml-1 opacity-70">{suffix}</span>}
      </div>
    );
  }
  
  if (readOnlyStyle) {
    return (
      <div className={`inline-flex items-center justify-center ${w} px-1 py-0.5 font-bold border-2 border-black bg-slate-100`}>
        <span>{displayValue}</span>
        {suffix && <span className="text-xs ml-1 opacity-70">{suffix}</span>}
      </div>
    );
  }

  return (
    <div className="inline-flex items-center">
      <input 
        type="number" name={name} value={value === undefined ? '' : value} onChange={onChange || (() => {})} readOnly={!onChange} placeholder="?" 
        className={`${w} bg-yellow-50 border-2 border-slate-400 text-center font-bold text-blue-900 shadow-inner focus:outline-none focus:border-blue-600 transition-colors px-1 py-0.5`} 
      />
      {suffix && <span className="text-xs ml-1 font-bold text-slate-600">{suffix}</span>}
    </div>
  );
};

// --- 6. PURE CALCULATION ENGINE ---
const calculateKillSheet = (wellData, pumpData, stringData, annulusData, kickData) => {
  const activeDisp = pumpData.selectedPump == 1 ? safeNum(pumpData.disp1) : safeNum(pumpData.disp2);
  const activeScr = pumpData.selectedPump == 1 ? safeNum(pumpData.scr1) : safeNum(pumpData.scr2);
  const activeSpm = pumpData.selectedPump == 1 ? safeNum(pumpData.spm1) : safeNum(pumpData.spm2);

  const mamwRaw = (safeNum(wellData.lotPressure) / ((safeNum(wellData.shoeTVD) || 1) * 0.052)) + safeNum(wellData.testMW);
  const mamw = safeNum(wellData.shoeTVD) > 0 ? roundDown(mamwRaw, 1) : 0;
  const initialMaasp = roundDown((mamw - safeNum(wellData.currentMW)) * safeNum(wellData.shoeTVD) * 0.052, 0);
  
  const kmwRaw = (safeNum(kickData.sidpp) / ((safeNum(wellData.wellTVD) || 1) * 0.052)) + safeNum(wellData.currentMW);
  const kmw = safeNum(wellData.wellTVD) > 0 ? roundUp(kmwRaw, 1) : 0;
  const finalMaasp = roundDown((mamw - kmw) * safeNum(wellData.shoeTVD) * 0.052, 0);

  const icp = roundNearest(safeNum(kickData.sidpp) + activeScr);
  const fcp = safeNum(wellData.currentMW) > 0 ? roundNearest(activeScr * (kmw / safeNum(wellData.currentMW))) : 0;

  const dpLength = Math.max(0, safeNum(wellData.wellMD) - safeNum(stringData.hwdpLength) - safeNum(stringData.dcLength));
  const dpVol = round1(dpLength * safeNum(stringData.dpCap));
  const hwdpVol = round1(safeNum(stringData.hwdpLength) * safeNum(stringData.hwdpCap));
  const dcVol = round1(safeNum(stringData.dcLength) * safeNum(stringData.dcCap));
  const stringVol = round1(dpVol + hwdpVol + dcVol);
  const stringStks = activeDisp > 0 ? roundNearest(stringVol / activeDisp) : 0;
  const stringTime = activeSpm > 0 ? roundNearest(stringStks / activeSpm) : 0;

  const ohLength = Math.max(0, safeNum(wellData.wellMD) - safeNum(wellData.shoeMD));
  const dcOhLength = Math.min(safeNum(stringData.dcLength), ohLength);
  const dcOhVol = round1(dcOhLength * safeNum(annulusData.ohDcCap));
  const dpHwdpOhLength = Math.max(0, ohLength - dcOhLength);
  const dpHwdpOhVol = round1(dpHwdpOhLength * safeNum(annulusData.ohDpCap));
  const ohVol = round1(dcOhVol + dpHwdpOhVol);
  const ohStks = activeDisp > 0 ? roundNearest(ohVol / activeDisp) : 0;
  const ohTime = activeSpm > 0 ? roundNearest(ohStks / activeSpm) : 0;

  const dpCsgLength = safeNum(wellData.shoeMD);
  const dpCsgVol = round1(dpCsgLength * safeNum(annulusData.csgDpCap));
  const annVol = round1(ohVol + dpCsgVol);
  const annStks = activeDisp > 0 ? roundNearest(annVol / activeDisp) : 0;
  const annTime = activeSpm > 0 ? roundNearest(annStks / activeSpm) : 0;

  const totalWellVol = round1(stringVol + annVol);
  const totalWellStks = activeDisp > 0 ? roundNearest(totalWellVol / activeDisp) : 0;
  const totalWellTime = activeSpm > 0 ? roundNearest(totalWellStks / activeSpm) : 0;

  const activeSurfVol = safeNum(pumpData.activeSurfaceVol);
  const activeSurfStks = activeDisp > 0 ? roundNearest(activeSurfVol / activeDisp) : 0;
  const activeSurfTime = activeSpm > 0 ? roundNearest(activeSurfStks / activeSpm) : 0;

  const totalActiveVol = round1(totalWellVol + activeSurfVol);
  const totalActiveStks = activeDisp > 0 ? roundNearest(totalActiveVol / activeDisp) : 0;
  const totalActiveTime = activeSpm > 0 ? roundNearest(totalActiveStks / activeSpm) : 0;

  const pressureDropTotal = icp - fcp;
  const pressureDropPerTenth = stringStks > 0 ? roundNearest(pressureDropTotal / 10) : 0;
  const pressureDropPer100 = stringStks > 0 ? round1((pressureDropTotal * 100) / stringStks) : 0;

  const chart = [];
  if (stringStks > 0) {
    for (let i = 0; i <= 10; i++) {
      let currentStrokes = i === 10 ? stringStks : roundNearest(i * (stringStks / 10));
      let currentPressure = i === 10 ? fcp : roundNearest(icp - (i * (pressureDropTotal / 10)));
      chart.push({ step: i, strokes: currentStrokes, pressure: currentPressure });
    }
  }

  return { 
    kmw, mamw, initialMaasp, finalMaasp, icp, fcp, 
    dpLength, dpVol, hwdpVol, dcVol, stringVol, stringStks, stringTime,
    dcOhLength, dcOhVol, dpHwdpOhLength, dpHwdpOhVol, ohVol, ohStks, ohTime,
    dpCsgLength, dpCsgVol, annVol, annStks, annTime,
    totalWellVol, totalWellStks, totalWellTime,
    activeSurfVol, activeSurfStks, activeSurfTime,
    totalActiveVol, totalActiveStks, totalActiveTime,
    pressureDropTotal, pressureDropPerTenth, pressureDropPer100, chart 
  };
};

const TestModeManager = ({ formatTime, userRole, setActiveTab }) => {
  const [testPhase, setTestPhase] = useState('idle'); // idle, active, results, review-sheet
  const [currentScenario, setCurrentScenario] = useState(null);
  
  // Isolated exam states
  const [testWell, setTestWell] = useState({});
  const [testString, setTestString] = useState({});
  const [testAnnulus, setTestAnnulus] = useState({});
  const [testPump, setTestPump] = useState({});
  const [testKick, setTestKick] = useState({});
  const [testStudentData, setTestStudentData] = useState({});
  
  const [examGrade, setExamGrade] = useState(null);
  const [examTime, setExamTime] = useState(0);

  // Auto-calculating correct engine for grading
  const correctCalc = useMemo(() => {
    if (!currentScenario) return {};
    return calculateKillSheet(currentScenario.well, currentScenario.pump, currentScenario.string, currentScenario.annulus, currentScenario.kick);
  }, [currentScenario]);

  useEffect(() => {
    let interval;
    if (testPhase === 'active') interval = setInterval(() => setExamTime(t => t + 1), 1000);
    return () => clearInterval(interval);
  }, [testPhase]);

  const startExam = (scId) => {
    const sc = EXAMPLES.find(e => e.id === scId);
    setCurrentScenario(sc);
    setTestWell({}); setTestString({}); setTestAnnulus({}); setTestPump({ selectedPump: 1 }); setTestKick({}); setTestStudentData({});
    setTestPhase('active');
    setExamTime(0);
  };

  const handleTestInput = (setter) => (e) => {
    let val = e.target.value;
    if (e.target.type === 'number') val = val === '' ? '' : parseFloat(val);
    setter(prev => ({ ...prev, [e.target.name]: val }));
  };
  const handleTestStudentInput = (e) => setTestStudentData(prev => ({ ...prev, [e.target.name]: e.target.value }));

  const submitTestExam = () => {
    let correctCount = 0; let wrongCount = 0; let details = [];

    const checkField = (name, section, expected, actual, tolerance = 0, isString = false) => {
      let isCorrect = false;
      if (isString) {
        isCorrect = String(actual || '').trim().toLowerCase() === String(expected || '').trim().toLowerCase();
      } else {
        const sVal = Number(actual); const eVal = Number(expected);
        if (actual === undefined || actual === '' || isNaN(sVal)) isCorrect = false;
        else isCorrect = Math.abs(sVal - eVal) <= tolerance;
      }
      if (isCorrect) correctCount++; else wrongCount++;
      
      const formatName = name.replace(/([A-Z])/g, ' $1').replace(/^./, str => str.toUpperCase());
      details.push({ name: formatName, section, expected, actual, isCorrect });

        checkField(`stepPressure_${i}`, 'Kill Sheet: Step Down Chart', pt.pressure, testStudentData[`stepPressure_${i}`], 5);

You must extract and enter the data into the <strong>Data Entry Form</strong>, and then utilize those metrics to successfully solve the <strong>Master Kill Sheet</strong>. Your performance across both sections will be graded simultaneously.</p>

        </div>
        
        <MasterKillSheetSection wellData={currentScenario.well} pumpData={currentScenario.pump} stringData={currentScenario.string} annulusData={currentScenario.annulus} kickData={currentScenario.kick} studentData={testStudentData} handleStudentInput={()=>{}} calc={correctCalc} isCalculated={true} userRole="student" hideSubmit={true} showOutput={false} />
      </div>
    );
  }

  return null;
};
